134 // Verificar si el resultado es de tipo mysqli_result

136 if (mysqli_num_rows($resultado) > 0) {
137 while ($fila = mysqli_fetch_assoc($resultado)) {

144 echo "No se encontraron resultados.<br>";

147 // Si el resultado es TRUE y no un mysqli_result, significa que la operación fue exitosa pero no es de tipo SELECT
